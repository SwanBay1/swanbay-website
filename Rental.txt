  <script>
    // If you don't set an image for a property, this placeholder will be used.
    // File: Website/Assets/Placeholder.png
    const PLACEHOLDER = "Assets/Placeholder.png";

    // ✅ Easy to add/remove listings:
    // Just add/remove objects in this array.
    // status must be one of:
    // "Currently available" | "Accepting reservations" | "Let"
    const RENTAL_PROPERTIES = [
      {
        title: "Baytree House - Double bed - front of house room",
        area: "125 Rhondda St.",
        price: "£550 pcm",
        beds: 1,
        baths: 2,
        status: "Currently available",
        href: "BaytreeUFroom.html",
        image: "Assets/uf1.jpeg"
      },
      {
        title: "Baytree House - Double bed - top floor room",
        area: "125 Rhondda St.",
        price: "£450 pcm",
        beds: 1,
        baths: 2,
        status: "Currently available",
        href: "BaytreeUMroom.html",
        image: "Assets/125RhonddaSt/UpstairsMiddle/1UM.jpg"
      }
          ];

    function escapeHtml(s){
      const d = document.createElement("div");
      d.textContent = s;
      return d.innerHTML;
    }

    function bedLabel(n){
      if (n === 0) return "Studio";
      if (n === 1) return "1 bed";
      return `${n} beds`;
    }

    function statusClass(status){
      const s = (status || "").toLowerCase();
      if (s.includes("currently available")) return "available";
      if (s.includes("accepting reservations")) return "reservations";
      return "unavailable";
    }

    const list = document.getElementById("rentalsList");

    function render(){
      list.innerHTML = RENTAL_PROPERTIES.map(p => {
        const title = escapeHtml(p.title);
        const area  = escapeHtml(p.area);
        const price = escapeHtml(p.price);
        const href  = escapeHtml(p.href || "#");
        const img   = (p.image && p.image.trim().length) ? escapeHtml(p.image) : PLACEHOLDER;

        const statusText = escapeHtml(p.status || "Let");
        const sClass = statusClass(p.status);

        return `
          <a class="row" href="${href}" aria-label="Open ${title}">
            <div class="thumb" aria-hidden="true">
              <img src="${img}" alt="">
            </div>

            <div class="row-main">
              <h3 class="row-title">${title}</h3>
              <div class="row-meta">
                <span>${area}</span>
                <span aria-hidden="true">•</span>
                <span>${bedLabel(p.beds)} • ${p.baths} bath${p.baths === 1 ? "" : "s"}</span>
              </div>
            </div>

            <div class="row-side">
              <div class="status ${sClass}">${statusText}</div>
              <div class="price">${price}</div>
            </div>
          </a>
        `;
      }).join("");
    }

    render();
  </script>
